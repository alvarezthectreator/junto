🔌 API ENDPOINTS READY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Authentication
  POST   /api/auth/login                        → Login with phone number
  GET    /api/auth/verify                       → Verify session

Users & Profiles
  GET    /api/users/:userId                     → Get user
  GET    /api/users/:userId/profile             → Get full profile
  PUT    /api/users/:userId/profile             → Update profile
  GET    /api/users/search                      → Search users
  GET    /api/users/travel-mode/:city           → Get users in travel mode

Events
  GET    /api/events                            → List events (with filters)
  GET    /api/events/:eventId                   → Get event details
  POST   /api/events                            → Create event
  PUT    /api/events/:eventId                   → Update event
  DELETE /api/events/:eventId                   → Delete event
  GET    /api/events/host/:hostId               → Host's events

Applications
  POST   /api/applications                      → Apply to event
  GET    /api/applications/user/:userId         → User's applications
  GET    /api/applications/event/:eventId       → Event's applications
  PUT    /api/applications/:id/status           → Accept/reject
  DELETE /api/applications/:id                  → Withdraw

Messaging
  POST   /api/messages                          → Send message
  GET    /api/messages/conversations/:userId    → Get conversations
  GET    /api/messages/:conversationId          → Get conversation history
  PUT    /api/messages/:id/read                 → Mark as read
  DELETE /api/messages/:id                      → Delete message

Nearby Mode
  GET    /api/nearby/:userId                    → Get nearby users
  POST   /api/nearby/swipe                      → Swipe on user
  GET    /api/nearby/:userId/matches            → Get matches
  GET    /api/nearby/:userId/history            → Swipe history

Safety Centre
  GET    /api/safety/:userId/contacts           → Get trusted contacts
  POST   /api/safety/:userId/contacts           → Add contact
  PUT    /api/safety/:contactId                 → Update contact
  DELETE /api/safety/:contactId                 → Delete contact
  POST   /api/safety/:userId/sos                → Trigger SOS alert
  GET    /api/safety/:userId/blocked            → Get blocked users
  POST   /api/safety/:userId/block/:blockedId   → Block user
  DELETE /api/safety/:userId/block/:blockedId   → Unblock user
  POST   /api/safety/:userId/report/:reportedId → Report user

Notifications
  GET    /api/notifications/:userId             → Get notifications
  PUT    /api/notifications/:id/read            → Mark as read
  DELETE /api/notifications/:id                 → Delete notification
